1 /**
  2  * This function will create the view for the products that are low in stock
  3  * @returns {HTMLElement} a div
  4 */
  5 function createLowestInStockView() {
  6     var lowest = theLowestInStock(5);
  7     var lowestContainer = document.createElement("div");
  8     lowestContainer.className = "lowest-in-stock-container";
  9     var h1 = document.createElement("h1");
 10     h1.id = "product-low-in-stock";
 11     lowestContainer.appendChild(h1);
 12     for (i in lowest) {
 13         lowestContainer.appendChild(createLowestInStockDiv(lowest[i].name, lowest[i].stock));
 14     }
 15     return lowestContainer;
 16 }
