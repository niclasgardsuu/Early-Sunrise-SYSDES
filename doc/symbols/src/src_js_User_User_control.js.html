 18 /**

 48 /**
 49  * Control the view for the log in for staff and manager
 50 */
 51 function logInStaff() {
 52 
 53     document.getElementById("product-info-hide").checked = false;
 54     $("#loginDisplay").html("");
 55     $("#loginDisplay").html(createSpanEvent("logOut","cursor","","logOut()") +
 56                             createSpanEvent("manager","cursor","","createManagerView()")+
 57                             createSpanEvent("bartender","cursor","","createBartenderView()")+
 58                             createSpanEvent("notify-security", "cursor", "", "rickRolled()"));
 59 
 60     if (theLowestInStock(5).length != 0) {
 61         alertBox(getString("less-than-five-warning"), 5000);
 62     } 
 63 
 64     if(OrderDB.all_orders.length != 0) {
 65         alertBox(getString("order-notification"), 5000);
 66     }
 67 
 68     updateViewUser();
 69 }

 88 /**
 89  *  Log the user out from the website and reset the view to the customer veiw
