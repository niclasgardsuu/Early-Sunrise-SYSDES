 11 /**
 12  * This function will remove a product from the database and update the view 
 13  * @returns {undefined} if remove did not work return null
 14 */
 15 function removeProductFromMeny() {
 16     var id = document.getElementById("product-info-articleid").textContent;
 17     var category = document.getElementById("product-info-category").textContent;
 18     for (var i = 0; i < drunk[category].length; i++) {
 19         if (drunk[category][i].articleid == id) {
 20             drunk[category].splice(i,1);
 21             document.getElementById("product-info-hide").checked = false;
 22             updateViewProducts(category);
 23             return;
 24         }
 25     } 
 26     return null;
 27 }

 29 /**
 30  * This function will check if the parameters are valid
 31  * @param {string} articleid the article id of the product
 32  * @param {string} category the main category which the product is in
 33  * @param {Array} cannotBeEmpty the array with all values that can't be empty
 34  * @returns {boolen} true if all checks passed else false
 35 */
 36 function checkAddProductToMeny(articleid, category, cannotBeEmpty) {
 37     var msg = "";
 38     var passed = true;
 39 
 40     for (var i = 0; i < cannotBeEmpty.length; i++ ) {
 41         if (cannotBeEmpty[i] == "") {
 42             passed = false;
 43             msg = getString("product-manager-input-error");
 44         }
 45     }
 46 
 47     if (findProductByID(articleid) != null){
 48         msg += getString("product-manager-id-error");
 49         passed = false;
 50     }
 51 
 52     if (category != dict.mainCategory[0] && category != dict.mainCategory[1] 
 53         &&  category != dict.mainCategory[2] && category != dict.mainCategory[3]) {
 54         msg += getString("product-manager-category-error");
 55         passed = false;
 56     }
 57 
 58     if (!passed) alertBox(msg, 3000);
 59 
 60     return passed;
 61 }

 63 /**
 64  * This function will add a product to the database
 65 */
 66 function addProductToMeny() {
 67     
 68     var articleid = document.getElementById("product-manager-articleid-i").value;
 69     var name = document.getElementById("product-manager-name-i").value;
 70     var name2 = document.getElementById("product-manager-name2-i").value;
 71     var pricewithvat = document.getElementById("product-manager-pricewithvat-i").value;
 72     var volume = document.getElementById("product-manager-volume-i").value;
 73                     
 74     var productgroup = document.getElementById("product-manager-productgroup-i").value;
 75     var serves = document.getElementById("product-manager-serves-i").value;
 76     var origin = document.getElementById("product-manager-origin-i").value;
 77     var origincountry = document.getElementById("product-manager-origincountry-i").value;
 78     var producent = document.getElementById("product-manager-producent-i").value;
 79     var alcoholcontent = document.getElementById("product-manager-alcohol-i").value;
 80     var img = "./img/" + document.getElementById("product-manager-img-i").value;
 81     var stock = document.getElementById("product-manager-stock-i").value;
 82     var category = document.getElementById("product-manager-main-category-i").value;
 83 
 84     var cannotBeEmpty = [articleid, name, pricewithvat, volume, productgroup, serves, origincountry, producent, alcoholcontent];
 85     var passed = checkAddProductToMeny(articleid, category, cannotBeEmpty);
 86     if (!passed) return;
 87 
 88     product = {
 89         "articleid": articleid,
 90         "name": name,
 91         "name2": name2,
 92         "pricewithvat": pricewithvat,
 93         "volume": volume,
 94         "productgroup": productgroup,
 95         "serves": serves,
 96         "origin": origin,
 97         "origincountry": origincountry,
 98         "producent": producent,
 99         "alcoholcontent": alcoholcontent,
100         "img": img,
101         "stock": stock,
102     }
103     drunk[category].push(product);
104     $("#product-manager-success-msg").text(name + getString("product-manager-success-msg")).fadeIn();     
105     setTimeout(function() { $("#product-manager-success-msg").fadeOut(); }, 3000);
106 }

147 /**
148  *  Create all products within a category that has been filtered and print them out 
149  * @param {string} name the name of the product
150  * @param {string} category the category where the product belongs to
151  * @returns {HTMLElement} true if the addition is successful, otherwise false
152 */
153 function createProductsByFilter(filterId, category) {
154     var noFilter = true;
155     document.getElementById("product-window").textContent = "";
156     for (var i = 0; i < filterId.length; i++) {
157         //check if the filter is checked
158         if (document.getElementById(filterId[i]).checked) {
159             noFilter = false;
160             var productgroup = filterId[i].replace(/\-/g, ' ');
161             var bevType = drunk[category];
162             var productWindow = document.getElementById("product-window");
163             for (var j = 0; j < bevType.length; j++) {
164                 //Print only out the products that has the same productgroup
165                 if (bevType[j].productgroup == productgroup) {
166                     productWindow.appendChild(createProductContainer(bevType[j].name, bevType[j].pricewithvat, bevType[j].img, bevType[j].articleid, category));
167                 }
168             }
169 
170         }
171     }
172 
173     //if no filter is checked we print out all the products in the category
174     if (noFilter) {
175         createProductsByCategory(category);
176     }
177     updateViewClasses("classes");
178 }

207 
208     var filterContainer = document.getElementById("filter-window");
209     for(var i = 0; i < filter.length; i++) {
210         filterContainer.appendChild(createFilterView(category, filterId, filterId[i]));
211     }

240  * @param {string} msg the message that will be shown
241  * @param {string} time the time in ms that the alert box will be shown
242 */
243 function alertBox(msg, time) {
244     var container = document.getElementById("alert-box-container");

246

262 /** 
263  * Will display a big window containing information about a product
264  * @param {number} id the id of the product
265  * @param {string} category the category which the product belongs to
266 */
267 function showProductInfo(id, category) {
268     var product = findProductByID(id);
269     document.getElementById("product-info-hide").checked = true;
270 
271     var productContainer = document.getElementById("product-info-container-id");
272 
273     var origin = product.origincountry;
274     if (product.origin != "") { 
275         origin += ", " + product.origin;
276     }
277 
278     var allergens = getAllergens(id);
279     //check if there are no allergens for the product
280     if (allergens == null) {
281         allergens = "inga";
282     } 
283 
284     productContainer.innerHTML = "";    
285     productContainer.innerHTML = createShowProductBaseView(product);
286 
287     var containerRight = document.getElementById("product-info-container-right");
288 
289     if(modelData["credentials"] == 0) {
290         containerRight.innerHTML = createProductInfoView(product, origin, allergens, category) + createStaffInfoView(product);
291 
292         document.getElementById("product-manager-stock").addEventListener("change", changeStock.bind(null, id, null));
293 
294         var refill = document.getElementById("product-manager-refill");
295         refill.addEventListener("click", changeStock.bind(null, id, 24));
296         refill.addEventListener("click", function () {
297             document.getElementById("product-manager-stock").value = parseInt(document.getElementById("product-manager-stock").value) + 24;
298         });
299         document.getElementById("product-manager-remove-product").addEventListener("click", removeProductFromMeny);
300         document.getElementById("product-manager-price-i").addEventListener("change", changePrice.bind(null, product.articleid, category));
301     } else {
302         containerRight.innerHTML = createProductInfoView(product, origin, allergens, category);
303     }
304 
305     var temp = {
306                 execute : addToCart.bind(null, id, 1),
307                 unexecute: removeFromCart.bind(null, id, 1)
308             }
309     
310     document.getElementById("product-buy-id").addEventListener("click", doit.bind(null, temp));
311     updateView();
312 }

314 /** 
315  * Updates the shopping cart view to display the current state of the shopping cart
316 */
317 function updateShoppingCartView() {
318     var shoppingBottom = document.getElementById("shopping-cart-container-bottom");
319     shoppingBottom.textContent = "";
320 
321     for(var i = 0; i < OrderDB.cart.productId.length; i++) {
322         var id = OrderDB.cart.productId[i];
323         var productDiv = createShoppingCartDiv(id ,OrderDB.cart.productAmount[i]);
324         shoppingBottom.appendChild(productDiv);
325         var removeButton = document.getElementById(id+"-cart-button");  
326 
327         var temp = {
328                 execute: removeFromCart.bind(null, id, 1),
329                 unexecute : addToCart.bind(null, id, 1)
330             }
331 
332         removeButton.addEventListener("click", doit.bind(null, temp));  
333     }
334 
335     if (OrderDB.cart.productId.length == 0) {
336         shoppingBottom.insertAdjacentHTML("beforeend", createEmptyShoppingCart());
337     }
338 }
