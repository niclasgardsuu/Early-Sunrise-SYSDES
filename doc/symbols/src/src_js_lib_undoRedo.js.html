1 // The data base for the UNDO-REDO mechanism is stored in two stacks
  2 // Both of these are empty to start with.
  3 //
  4 let undostack = [];
  5 let redostack = [];
  6 
  7 // The UNDO-manager consists of three functions. In this version they are
  8 // rudimentary, in that they don't check for an empty stack before trying.
  9 // Adding these checks is an easy task.
 10 //
 11 // The UNDO-manager requires that the functions are stored as objects with three
 12 // functions each: execute, unexecute and reexecute. Since the model is very
 13 // small here, the functins are also necessarily simple. With more complex
 14 // models, the functions for this will also grow.
 15 //
 16 // Note, however. that the undo-mechanism does not need to know anything
 17 // about the model or the functions themselves. It just executes the appropriate
 18 // function for the respective moments.
 19 //
 20 // ==========================================================================
 21 // the doit function executes the function and then stores the function object
 22 // on the UNDO-stack.
 23 //
 24 // Note that when a new function is executed, the REDO-stack has to be reset!
 25 //
 26 function doit(funcobj) {
 27     funcobj.execute();
 28     //remove the oldest function if the stack is larger than 10
 29     if(undostack.length == 10) {
 30         undostack.shift();
 31     }
 32     undostack.push(funcobj);
 33     redostack = [];
 34 }
 35 
 36 
 37 // ==========================================================================
 38 // The undoit first pops the function object from the UNDO-stack, then executes
 39 // UNDO-function and stores the function object on the REDO-stack.
 40 //
 41 function undoit() {
 42     if (undostack.length == 0) return;
 43     funcobj = undostack.pop();
 44     funcobj.unexecute();
 45     redostack.push(funcobj);
 46 }
 47 
 48 // ==========================================================================
 49 // The redoit function pops the function object from the REDO-stack, executes
 50 // the EXECUTE-function and then pushes the function object onto the UNDO-stack.
 51 //
 52 function redoit() {
 53     if (redostack.length == 0) return;
 54 
 55     funcobj = redostack.pop();
 56     funcobj.execute();
 57     undostack.push(funcobj);
 58 }
