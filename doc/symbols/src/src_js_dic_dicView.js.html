 12 /**
 13  * This function will update any images on the website to a version of the image that responds to the currently chosen language.
 14  */
 15 function updateViewPics() {
 16     pics = dict["pics"];
 17     for (idx in pics) {
 18         pic = pics[idx][0];
 19         alt = pics[idx][1];
 20         id = document.getElementById(pic);
 21         if(id != null) {
 22             id.src = getString(pic);
 23             id.alt = getString(alt);
 24         }
 25     }; 
 26 }

 28 /**
 29  *  This function will update all boxes that have an ID attribute that is equivalent to an ID in the dictionary in order to update the language
 30  * @param {string} id the id of the dictionary
 31  */
 32 function updateViewIds(id) {
 33     ids = dict[id];
 34     for (idx in ids) {
 35         key = ids[idx];
 36         id = document.getElementById(key);
 37         if(id != null) id.innerText = getString(key);
 38     };
 39 }
 40 
 41 /**
 42  *  This function will update all boxes that have a class attribute that is equivalent to a class in the dictionary in order to update the language
 43  * @param {string} id the id of the dictionary
 44  */
 45 function updateViewClasses(id) {
 46     var name = dict[id];
 47     for (i in name) {
 48         var temp = document.getElementsByClassName(name[i]);
 49         if (temp != null) {
 50             for (j in temp) {
 51                 temp[j].innerText = getString(name[i]);
 52             }
 53         }
 54     }
 55 }

 57 /**
 58  *  This function will update the current text in an html element to display the currently chosen language
 59  * @param {string} attribute the attribute that the html element will have
 60  * @param {string} id the id of the dictionary
 61  */
 62 function updateViewAttribute(attribute, id) {
 63     var ids = dict[id];
 64     for (i in ids) {
 65         var key = ids[i];
 66         var id = document.getElementById(key);
 67         if (id != null) id.setAttribute(attribute, getString(key));
 68     }
 69 }

 71 /**
 72  * This will only update main dictionary
 73  */
 74 function updateViewMain() {
 75     updateViewIds("main-ids");
 76     updateViewPics();
 77     updateViewClasses("classes");
 78 }
