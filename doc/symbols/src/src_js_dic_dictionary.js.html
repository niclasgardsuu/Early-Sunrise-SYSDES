398 // This function will return the appropriate string for each
399 // key. The language handling is made "automatic".
400 //
401 function getString(key) {
402     return dict[language][key];
403 }
404 
405 // This function is the simplest possible. However, in order
406 // to handle many different languages it will not be sufficient.
407 // The necessary change should not be difficult to implement.
408 //
409 // After each language change, we will need to update the view, to propagate
410 // the change to the whole view.
411 //
412 function changeLang(sel) {
413     language = sel.options[sel.selectedIndex].value;
414     updateView();
415 }
