1 /**
  2  *  Create an attribute element
  3  * @param {string} attr an attribute
  4  * @param {string} content a content
  5  * @returns {HTMLElement} an attribute element
  6 */
  7 function createAttribute(attr,content) {
  8     if (content == "") {
  9         return "";
 10     } else {
 11         return attr + "=" + content + " ";
 12     }
 13 }

 15 /**
 16  *  Create a span with an even element
 17  * @param {string} id an id
 18  * @param {string} classname a classname
 19  * @param {string} content a content
 20  * @param {string} _function a function
 21  * @returns {string} a string of a span with an even
 22 */
 23 function createSpanEvent(id, classname, content, _function) {
 24     idString = createAttribute("id", id);
 25     classString = createAttribute("class", classname);
 26     return "<span " + idString + classString + "onclick="+_function+">" + content + "</span>";
 27 }

113 
114     for (var i = 0; i < order.productId.length; i++) {
115         // Name | Quantity | Price | Button
116         var td1 = document.createElement("td");
117         var product = findProductByID(order.productId[i]);
118         td1.appendChild(document.createTextNode(product.name));
119 
120         var td2 = document.createElement("td");
121         td2.appendChild(document.createTextNode(order.productAmount[i]));
122 
123         var td3 = document.createElement("td");
124         td3.appendChild(document.createTextNode(product.pricewithvat));
125 
126         var tr = document.createElement("tr");
127         tr.appendChild(td1);
128         tr.appendChild(td2);
129         tr.appendChild(td3);
130         table.appendChild(tr);
131     }
132 
133     var totalClass = ["bartender-total", "", ""];
134     var totalData = ["", calculateAmount(order.productAmount), order.totalPrice];
135     var tr = document.createElement("tr");
136     for (var i = 0; i < 3; i++) {
137         var td = document.createElement("td");
138         td.className = totalClass[i];
139         td.appendChild(document.createTextNode(totalData[i]));
140         tr.appendChild(td);
141     }
142     table.appendChild(tr);
143 
144 
145     var ids = ["change-order-","remove-order-", "finish-order-"];
146     var fun = [revertOrder.bind(null, order.order_id), removeOrder.bind(null, order), finishOrder.bind(null,order.order_id)];
147 
148     var tr2 = document.createElement("tr");
149     for (var i = 0; i < 3; i++) {
150         var td = document.createElement("td");
151         var button = document.createElement("button");
152         button.id = ids[i] + order.order_id;
153         button.className = "bartender-button-n-"+i;
154         button.addEventListener("click", fun[i]);
155         td.appendChild(button);
156         tr2.appendChild(td);
157     }
158     table.appendChild(tr2);
159 
160     return table;
161 }
